import { Bounce, toast } from "react-toastify";
const tostifyConfig = {
  position: "top-center",
  autoClose: 5000,
  hideProgressBar: false,
  closeOnClick: true,
  pauseOnHover: true,
  draggable: true,
  progress: undefined,
  theme: "light",
  transition: Bounce,
};

export const notifySuccess = () => {
  toast.success("Back Online", tostifyConfig);
};

export const notifyError = () => {
  toast.error("Check Your Internet Connection", tostifyConfig);
};
